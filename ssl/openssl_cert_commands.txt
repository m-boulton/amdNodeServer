cd ssl
# generate private key
openssl genrsa -out amdServer.key 2048

# generate Public Key
openssl rsa -in amdServer.key -pubout -out amdServer_public.key

# generate certificate signing request (self signed)
openssl req -new -key amdServer.key -out amdServer.csr

# Verify self signed keys
openssl x509 -in amdServer.csr -out amdServer.crt -req -signkey amdServer.key -days 365 

# verify the by other company
openssl req -text -in amdServer.csr -noout -verify
